import React from 'react';
import { View, Text, Image, StyleSheet } from 'react-native';
import PropTypes from 'prop-types';
// ANTD
import { Button } from 'antd-mobile';
// MODULES
import { DEFAULT_AVATAR, DEFAULT_POST_IMAGE, fontConfig } from '../../modules/config';
import FollowButton from '../FollowButton';
import RegularText from '../RegularText';




// DOCS
// =========================
/**
 *
 * @namespace UserCard
 * @description A card to display a user Avatar, full name, and Follow/Unfollow button. Used in contructing the results for a UsersSearch component.
 * @prop targetUser {object} - the target user is the person displayed on the user card. 
 * @prop currentUser {object} - The currently logged in user. we use properties from this object to match against the target user and answer questions like " are these peope already friends?"
*/


class UserCard extends React.PureComponent {
  renderFollowButton = () => {

    const { targetUser, currentUser, incoming, follow } = this.props;
    
    if (targetUser._id === currentUser._id) {
      return null;
    }

    return (
      <FollowButton 
        follow={follow} 
        targetUser={targetUser} 
        incoming={incoming} 
      />
    );
  };
  renderName = () => {
    const { targetUser, navigation, currentUser, incoming, follow } = this.props;
    return (
      <RegularText
        onPress={()=>navigation.navigate('userProfile', { _id: targetUser._id })}
      >
        {`${targetUser.profile.firstName} ${targetUser.profile.lastName}`}
      </RegularText>
    );
  }
  render() {
    const { targetUser, currentUser, incoming, follow } = this.props;
    
    // set a default image source 
    let imageSource = require('../../../assets/avatar.jpg');
    
    // if user has a profile picture, show it 
    if (targetUser.profile && targetUser.profile.image) {
      imageSource = { uri: targetUser.profile.image }
    }

    return (
      <View style={styles.cardContainer}>
        <View style={styles.column}>
          <Image
            style={{width: 40, height: 40, borderRadius: 20}}
            alt="post-avatar"
            source={imageSource}
          />
        </View>
       <View style={[styles.column, {flex: 3, alignItems: 'flex-start'}]}>
        {this.renderName()}
        </View>
        <View style={[styles.column, { flex: 2 }]}>
          {this.renderFollowButton()}
        </View>
      </View>
    );
  }
}


// STYLES
// ===================================
const styles = StyleSheet.create({
  cardContainer: {
    minHeight: 55, 
    marginTop: 3, 
    display: 'flex', 
    flexDirection: 'row'
  },
  column: {
    flex: 1, 
    display: 'flex', 
    alignItems: 'center', 
    justifyContent: 'center'
  }
});


UserCard.propTypes = {
  targetUser: PropTypes.object,
  currentUser: PropTypes.object,
};

// Specifies the default values for props:
UserCard.defaultProps = {
  targetUser: null,
  currentUser: null,
};


export default UserCard;
