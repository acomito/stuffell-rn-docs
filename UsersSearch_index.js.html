/**
 *
 * @namespace UsersSearch
 * @description This is a component used in the Friend's Tab
 * @prop searchText {string} - this is a string used to search for friends in the datbase. it's stored in redux usually.
 */


const FriendsOfLink = ({ onClick }) => {
  return (
    <Text style={{ textAlign: 'right', marginTop: 10, fontSize: 17 }}>
      Or, see some {' '}
      <Text onClick={onClick}>
        friend recommendations
      </Text>
    </Text>
  )
};

class UsersSearch extends React.PureComponent {
  state = {
    showFriendsOf: false,
  };
  renderFriendsOfLink = () => {
    // 
    if (!this.state.showFriendsOf) {
      return <FriendsOfLink onClick={() => this.setState({ showFriendsOf: true })} />;
    }

    return (
      <Text style={{ textAlign: 'right', marginTop: 10, fontSize: 17 }}>
        Or, 
        <Text onClick={() => this.setState({ showFriendsOf: false })}>
          search
        </Text>
      </Text>
    );
  };
  render() {
    const { 
      screenProps, 
      navigation, 
      searchText, 
      onFriendSearchTextChange, 
      togglePostModal, 
      postModalOpen 
    } = this.props;
    const { user } = screenProps.data;

    return (
      <View style={{ flex: 1 }}>
        <View style={{height: 58}}>
          <SearchBar
              onChangeText={val => onFriendSearchTextChange(val)}
              inputStyle={{ backgroundColor: '#fff' }}
              containerStyle={{ width: SCREEN_WIDTH }}
              placeholder={"Search..."}
              lightTheme
            />
        </View>
        <ScrollView style={{ paddingRight: 8, paddingLeft: 8 }}>
          <View style={{ marginTop: 10 }}>
          <FriendsOfFriends 
            navigation={navigation} 
            searchText={searchText} 
            currentUser={user} 
          />
        </View>
          {!this.state.showFriendsOf && (
            <UsersSearchList 
              navigation={navigation} 
              currentUser={user}
              searchText={searchText} 
            />
          )}
        </ScrollView>
      </View>
    );
  }
}


// REDUX
// ============================
let mapStateToProps = ({ search, modals }) => {
  return {
    searchText: search.searchText_friends,
  };
};


// PROP TYPES
// ============================
UsersSearch.propTypes = {
  searchText: PropTypes.string
}

export default connect(mapStateToProps, actions)(UsersSearch);
