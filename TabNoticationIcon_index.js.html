import React from 'react';
import { View, StyleSheet } from 'react-native';
import { Feather } from '@expo/vector-icons';
import { graphql, compose } from 'react-apollo';
import { 
    GET_INCOMING_FOLLOWS, 
    HAS_UNREAD_NOTIFICATIONS, 
    HAS_UNREAD_BUYING, 
    HAS_UNREAD_SELLING 
} from '../../apollo/queries'


const POLL_INTERVAL = 60000
const s = StyleSheet.create({
    container: {
        position: 'relative', 
        display: 'flex', 
        alignItems: 'center', 
        justifyContent: 'center'
    },
    notification: {
        right: 0, 
        top: 6, 
        position: 'absolute', 
        width: 10, 
        height: 10, 
        borderRadius: 5, 
        backgroundColor: 'red'
    }
})

/**
 *
 * @namespace TabNoticationIcon
 * @description Used in the TabNavigator, usually, for the Friends and Activity tabs, which have some notification-like functionality. 
 * This icon will show up with a red dot when users have new messages. We should probably abstract this logic to the back end...
 * @prop incomingFollows {object} - an apollo query that holds an array of requests from other users who want to follow me (in other words, friend me)
 * @prop hasUnreadNotificationsData {object} - tells us if the current user has unread notifications.
 * @prop hasUnreadBuyingData {object} - the target user is the person displayed on the user card.
 * @prop hasUnreadSellingData {object} - the target user is the person displayed on the user card.
 * @prop type {string} - 
 * we can pass a 'type' prop into TabNoticationIcon to determine if 
 * this icon is used for the friends tab or the acvitity tab, and it's used 
 * in our lofic to display the dot or not.
*/


class TabNoticationIcon extends React.PureComponent {
    render(){
        const { tintColor, type, name, incomingFollows  } = this.props;

        // we use a switch statement to check if we should show a red circle/dot around the tab's icon.
        let highlightBadge = false;
        switch(type){
          case 'activity':
            // 
            highlightBadge =  (!this.props.hasUnreadSellingData.loading && this.props.hasUnreadSellingData.hasUnreadSelling.result) || (!this.props.hasUnreadBuyingData.loading && this.props.hasUnreadBuyingData.hasUnreadBuying.result) || (!this.props.hasUnreadNotificationsData.loading && this.props.hasUnreadNotificationsData.hasUnreadNotifications.result);
            break;
          case 'friends':
            if (!incomingFollows.loading) {
              highlightBadge = incomingFollows.incomingFollows.length && incomingFollows.incomingFollows.length > 0 ? true : false;
            }
            break;
          default:
            highlightBadge = false;
        }

        return (
            <View style={s.container}>
                <Feather 
                    name={name}
                    size={30} 
                    color={tintColor} 
                />
                {highlightBadge && <View style={s.notification} />}
            </View>
        )
    }
}


export default compose(
    graphql(GET_INCOMING_FOLLOWS,  { options: { pollInterval: POLL_INTERVAL }, name: 'incomingFollows'}),
    graphql(HAS_UNREAD_NOTIFICATIONS, { options: { pollInterval: POLL_INTERVAL }, name: 'hasUnreadNotificationsData'} ),
    graphql(HAS_UNREAD_BUYING, { options: { pollInterval: POLL_INTERVAL }, name: 'hasUnreadBuyingData'} ),
    graphql(HAS_UNREAD_SELLING, { options: { pollInterval: POLL_INTERVAL }, name: 'hasUnreadSellingData'} ),
  )(TabNoticationIcon);
